#include <iostream>
#include <map>
#include <queue>
using namespace std;

int main()
{
	int n, m;
	cin >> n >> m;

	map<int, int> sol;
	for (int i = 0; i < n; i++)
	{
		int x, y;
		cin >> x >> y;
		sol.insert(make_pair(x, y));
	}

	for (int i = 0; i < m; i++)
	{
		int u, v;
		cin >> u >> v;
		sol.insert(make_pair(u, v));
	}

	queue<int> q;
	q.push(1);
	int dp[101] = {0, 0,};
	while(!q.empty())
	{
		int current = q.front();
		q.pop();

		for (int i = 1; i <= 6; i++)
		{
			if (current + i > 100) continue;

			if (dp[current + i] != 0 && 1 + dp[current] >= dp[current + i]) continue;

			dp[current + i] = 1 + dp[current];

			int move = sol[current + i];
			if (move)
			{
				dp[move] = dp[move] == 0 ? dp[current + i] : min(dp[move], dp[current + i]);
				q.push(move);
			}
			else
			{
				q.push(current + i);
			}
		}
	}

	cout << dp[100] << '\n';

	return 0;
}
```
